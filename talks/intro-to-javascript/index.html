JavaScript is functional

functions are values

var sum = function(a, b) {
    return a + b;
};

var foo = sum;

assert( foo(2, 3) == 5 );

functions are values

var double = function(a) {
    return a * 2;
};

function twice(op, x) {
    return op(op(x));
};

assert( twice(double, 2) == 8 );

higher order functions

var array = [1, 2, 3];

var doubled = array.map(function(e) {
    return e * 2;
});

assert( equal(doubled, [2, 4, 6] );

var total = array.reduce(function(sum, e) {
    return sum + e;
}, 0);

assert( total == 6 );

Scope

lexical scope

var x = 3;
var y = 4;

(function a() {
    var y = 5;

    assert(x == 3);
    assert(y == 5);
}());

assert(x == 3);
assert(y == 4);

function-level scope

var y = 4;

for (var i = 0; i < 10; i += 1) {
    var y = i;
}

assert(y == 9);
assert(i == 10);

functions are closures

function outer() {
    var x = 3;

    return function() {
        x += 1
        return x;
    };
}
var inner = outer();

assert(typeof x == 'undefined');

assert(inner() == 4);
assert(inner() == 5);

encapsulation

define('mymodule', [], function() {
    var privateA, privateB;

    function privateFunc() {
        // ...
    }

    function publicFunc(a, b) {
        privateA = privateFunc(a);
        return privateB;
    }

    return {
        publicFunc: publicFunc
    };
});
define('counter', [], function() {
    function Counter() {
        var count = 0;

        function increment() {
            count += 1;
        }

        return {
            click: function() {
                increment();
                return count;
            }
        };
    };
    return Counter;
});
define('eg', [counter], function(Counter) {

    var count = Counter();

    count.click();
    count.click();

    assert( count.click() == 3 );

    assert(
      typeof count.increment == 'undefined'
    );

});

Events and Concurrency

JavaScript is single-threaded

var clicks = 0;

$('button').on('click', function() {
    clicks += 1;
});

watch out for code that blocks

var clicks = 0;

$('button').on('click', function() {
    clicks += 1;
});

while (clicks < 5) {
    // wait for more clicks...
}

// We never get here!
doNextThing();

correct implementation of click counter

var clicks = 0;

$('button').on('click', function() {
    clicks += 1;

    if (clicks >= 5) {
        doNextThing();
    }
});

all IO is asynchronous

var finished = false;

$.getJSON('/feed').done(function(feed) {
    display(feed.count +" unread items");

    finished = true;
});

assert( finished === false );

RSS reader

var feeds = [
    'http://www.reddit.com/.rss'
];

app.use(express.bodyParser());

app.post('/feeds', function(req, res) {
    feeds.push(req.body.url);
    var id = feeds.length - 1;
    res.redirect('/feeds/'+ id);
});
app.get('/feeds/:id', function(req, res) {
    var url = feeds[req.params.id];

    http.read(url)
    .then(function(data) {
        var xml  = data.toString('utf-8');

        var list = parseTitles(xml)
                   .join('<li>');

        res.send('<ul><li>'+list+'</ul>');
    });
});
function parseTitles(data) {
    var titles = [];
    data.replace(/<title>(.+?)<\/title>/ig,
        function(_, title) {
            titles.push(title);
        }
    );
    return titles;
}
