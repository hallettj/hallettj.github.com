JavaScript is functional

functions are values

var sum = function(a, b) {
    return a + b;
};

var foo = sum;

assert( foo(2, 3) == 5 );

functions are values

var double = function(a) {
    return a * 2;
};

function twice(op, x) {
    return op(op(x));
};

assert( twice(double, 2) == 8 );

higher order functions

var array = [1, 2, 3];

var doubled = array.map(function(e) {
    return e * 2;
});

assert( doubled[2] == 6 );

var total = array.reduce(function(sum, e) {
    return sum + e;
}, 0);

assert( total == 6 );

this

refers to the receiver

var obj = {
    func: function() {
        return this;
    }
}

assert( obj.func() == obj );

except when it doesn't

var obj = {
    func: function() {
        return this;
    }
}

var func = obj.func;

assert( func() != obj );

four invocation patterns

1. function invocation

var obj = {
    func: function() {
        return this;
    }
}

var func = obj.func;

// `this` refers to the global object
assert( func() == window );

2. method invocation

var obj = {
    func: function() {
        return this;
    }
}

// `this` refers to the receiver
assert( obj.func()    == obj );
assert( obj['func']() == obj );

3. constructor invocation

var obj = {
    func: function() {
        return this;
    }
}

// `this` refers to a new object
assert( new obj.func() != obj );
assert( new obj.func() != window );

4. apply invocation

var obj = {
    func: function() {
        return this;
    }
}

var otherFunc = function() { return this; };

// `this` refers to the given argument
assert( otherFunc.apply(obj) == obj );
assert( otherFunc.call(obj)  == obj );

apply() and call()

function sum(a, b) {
    return a + b;
}

assert( sum.apply(null, [1, 2]) == 3 );
assert( sum.call(null, 1, 2)    == 3 );

JavaScript is Object-Oriented

object literals

var obj = {
    foo: 1,
    bar: 2,
    sum: function(a, b) {
        return a + b;
    }
};

accessing properties

assert( obj.foo == 1 );

for (var prop in obj) {
    assert( typeof prop == 'string' );
    assert( prop in obj );

    obj[prop];
}

square bracket notation

assert( obj['foo'] == 1 );

var s = 'baz';
obj[s] = 3;

assert( obj.baz == 3 );

undefined properties

assert( typeof obj.whatev == 'undefined' );

deleting properties

delete obj.foo

assert( typeof obj.foo == 'undefined' );
assert( !('foo' in obj) );

prototypes

var proto = { foo: 1 };

function Proto() {};
Proto.prototype = proto;

var obj = new Proto();

assert( obj.foo == 1 );
assert( !obj.hasOwnProperty('foo') );

proto.foo = 2;
assert( obj.foo == 2 );

prototypes made easier

var proto = { foo: 1 };
var obj = Object.create(proto);

assert( obj.foo == 1 );
assert( !obj.hasOwnProperty('foo') );

proto.foo = 2;
assert( obj.foo == 2 );

Resig's Class implementation

var Button = Class.extend({
    init: function(interval) {
        this.interval = interval;
        this.lastPushed = new Date();
    },

    push: function() {
        if (this._intervalElapsed()) {
            this._badThings();
        } else {
            this.lastPushed = new Date();
        }
    },

    _intervalElapsed: function() {
        var now = new Date();
        return (now - this.lastPushed) < this.interval;
    },

    _badThings: function() {
        throw "You waited too long";
    }
});

subclassing

var EventedButton = Button.extend({
    init: function(interval) {
        this._super(interval);
        this._startCount();
    },

    push: function() {
        clearTimeout(this.timeout);
        this._startCount();
    },

    _startCount: function() {
        var self = this;
        this.timeout = setTimeout(function() {
            self._badThings();
        }, this.interval);
    }
});

var mark = new Date(), now;

Top 5 reasons my code does not work in IE

1. Extra comma in object literal

var myObj = {
    foo: 1,
    bar: 2,
}
  • Tolerated by most browsers, syntax error in IE.
  • Error: "Expected identifier, string or number"
  • Use syntax checker like JSLint <jslint.com>.

2. Use of Array#indexOf()

if ([1, 2, 3].indexOf('foo') > -1) {
    return true;
}
  • Error: "is not a method"
  • Use a framework replacement such as $.inArray().
  • Or define this method yourself.

3. Line break between return and return value

function getVals() {
    return
    {
        foo: 1, bar: 2
    };
}
  • Return value is undefined.
  • Be obsessive about line breaks and semicolons.
  • Use a syntax checker like JSLint.

4. Invalid arguments to a native function

$('#someDiv').css('width', 'NaNpx');
  • Error: "invalid argument"
  • IE is very strict about argument validation and number of arguments.

5. Treating a built-in function like a function

var oldWrite = document.write;
document.write = function() {
    if (!$.isReady) {
        return oldWrite.apply(document,
                              arguments);
    }
};
  • Error: "is null or not an object"
  • Native functions in IE do not inherit from Function.prototype.

  • Borrow functions from Function.prototype.

    return Function.prototype.apply.call(oldWrite, document, arguments);

  • Use framework methods for checking type, such as $.isFunction().
