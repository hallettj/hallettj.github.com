# Makefile

.PHONY: all clean

babel     := node_modules/.bin/babel
src_files := $(shell find . -name '*.js.flow' -not -path './node_modules/*')
out_files := $(patsubst %.js.flow,%.js,$(src_files))

all: $(out_files)

%.js: %.js.flow
    $(babel) $< --out-file $@

clean:
    rm -f $(out_files)
# Makefile

.PHONY: all clean watch

src := $(shell find src)

all: $(src)
  $(babel) src \
    --out-dir build \
    --source-maps

watch:
  $(babel) src \
    --out-dir build \
    --source-maps \
    --watch

clean:
    rm -rf build
